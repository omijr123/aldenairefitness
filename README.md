### Code Example: Timer Implementation
```java
public class WorkoutTimer {
    private CountDownTimer countDownTimer;
    private long timeLeftInMillis;
    private TimerListener listener;
    
    public interface TimerListener {
        void onTick(long millisUntilFinished);
        void onFinish();
    }
    
    public void startTimer(long durationInMillis) {
        countDownTimer = new CountDownTimer(durationInMillis, 1000) {
            @Override
            public void onTick(long millisUntilFinished) {
                timeLeftInMillis = millisUntilFinished;
                if (listener != null) {
                    listener.onTick(millisUntilFinished);
                }
                
                // Voice cues at specific intervals
                if (millisUntilFinished < 4000 && millisUntilFinished > 3000) {
                    speak("3 seconds left!");
                }
            }
            
            @Override
            public void onFinish() {
                timeLeftInMillis = 0;
                if (listener != null) {
                    listener.onFinish();
                }
                speak("Time's up! Great job!");
            }
        }.start();
    }
    
    public void pauseTimer() {
        if (countDownTimer != null) {
            countDownTimer.cancel();
        }
    }
    
    public void resumeTimer() {
        startTimer(timeLeftInMillis);
    }
    
    private void speak(String text) {
        // Text-to-speech implementation
        if (textToSpeech != null) {
            textToSpeech.speak(text, TextToSpeech.QUEUE_FLUSH, null, null);
        }
    }
}
```
